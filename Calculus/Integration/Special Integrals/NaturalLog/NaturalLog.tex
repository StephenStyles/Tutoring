\title{Integral of $\ln(x)$}

\author{Stephen Styles}


\begin{document}

\maketitle

\begin{align*}
\int \ln(x) \, dx 
\end{align*}
Let $u = \ln(x)$, then $\displaystyle{du = \frac{1}{x}\, dx}$\\

Let $dv = dx$, then $v = x$
\begin{align*}
\int \ln(x) \, dx &= u\,v -\int v \, du\\[1.5ex]
&= x\ln(x) - \int \frac{x}{x} \, dx\\[1.5ex]
&= x\ln(x) - \int 1 \, dx\\[1.5ex]
&= x \ln (x) - x + C
\end{align*}

Therefore:
\begin{align*}
\int \ln(x) \, dx &= x \ln (x) - x + C
\end{align*}
