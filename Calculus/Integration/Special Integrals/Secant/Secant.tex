\title{Integral of $\sec(x)$}

\author{Stephen Styles}


\begin{document}

\maketitle

\begin{align*}
\int \sec(x) \, dx &= \int \sec(x)\bigg(\frac{\sec(x)+ \tan(x)}{\sec(x)+\tan(x)}\bigg) \, dx\\
&= \int \frac{\sec^2(x)+ \sec(x)\tan(x)}{\sec(x)+\tan(x)} \, dx\\
\end{align*}
Let $u = \sec(x)+\tan(x)$, then $du = \sec(x)\tan(x)+ \sec^2(x) \, dx$

\begin{align*}
\int \sec(x) \, dx &= \int \frac{1}{u} \, du\\
&= \ln\big| u \big| + C\\
&= \ln\big| \sec(x) + \tan(x) \big| + C
\end{align*}
